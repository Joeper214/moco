  FIRSTNAME Varchar(50) NOT NULL,
  MIDDLENAME Varchar(30) NOT NULL,
  LASTNAME Varchar(30) NOT NULL,
  COURSE Varchar(100),
  CURRICULUM Integer,
  ACADSTATUS Varchar(100),
  SCHOLARSHIP Varchar(50),
  SEX SEXES DEFAULT 'Female',
  CIVILSTATUS Varchar(20),
  RELIGION Varchar(100),
  NATIONALITY Varchar(20) DEFAULT 'Filipino',
  BIRTHDAY Timestamp,
  BIRTHPLACE Varchar(100),
  ADD_CITY Varchar(100),
  ADD_PROV Varchar(100),
  TRIBE Varchar(50),
  HIGHSCHOOLGRADUATED Varchar(100),
  SCHOOLLASTATTENDED Varchar(100),
  DATESCHOOLLASTATTENDED Timestamp,
  TERMADMITTED SEMESTERS DEFAULT '1st semester',
  YEARADMITTED Integer,
  DATEADMITTED Timestamp DEFAULT 'NOW',
  TORCONTROLNUM Integer,
  TORREMARKS Varchar(200),
  YEARLEVEL Varchar(15) DEFAULT '1st year' NOT NULL,
  ADMITTEDBY Varchar(70),
  "LOCK" Integer DEFAULT 0,
  DATEGRADUATED Timestamp,
  GUARDIAN Varchar(100),
  REMARKS Varchar(500),
  ENTRYDATA Varchar(500),
  INCLUDECOURSE Integer DEFAULT 1,
  INCLUDEDATEGRAD Integer DEFAULT 1,
  ASSESSED Integer DEFAULT 0,
  ENTRANCEDATA Varchar(200),
  SPECIALORDER Varchar(100),
  MOBILENUMBER Varchar(20) DEFAULT NULL,
  BILLINGPROFILE Varchar(100),
  AVERAGE Float,
  USER_NAME Varchar(50),
  EMAIL_PASS Varchar(50),
  EMAIL_USER Varchar(50),
  GUARDIANADDRESS Varchar(100),
  SIGNATURE Blob sub_type 0,
  GUARDIANNUMBER Varchar(30),
  PICTURE Blob sub_type 0,
  LASTYEARATTENDED Integer DEFAULT 0,
  LASTTERMATTENDED Varchar(20) DEFAULT NULL,
  ACCOUNTRENEWED Integer DEFAULT 0,
  DATECORPRINTED Timestamp DEFAULT NULL,
  DATEONLINENROLL Timestamp DEFAULT NULL,
  GENDER Varchar(20),
  ALLOWONLINE Integer DEFAULT 0,
  ENTRANCEEXAMTAKEN Varchar(30),
  ENTRANCEEXAMDATE Date,
  ENTRANCEEXAMGENRATING Integer,
  ADMISSIONSTATUSONENTRANCE Varchar(50),
  CORPRINTEDBY Varchar(100) DEFAULT NULL,

  MOTHERNAME Varchar(100),
  FAMILYANNUALGROSSINCOME Numeric(10,2),

  GUARDIANRELATIONSHIP Varchar(50),
  HSHONORRECEIVED Varchar(50),
  MSUSIDNUM Bigint NOT NULL,
  REGISTRARCLERKNOTES Varchar(200),
  TORGRADNOTE Varchar(100),
  PRIMARY KEY (IDNUMBER)
);

UPDATE RDB$RELATION_FIELDS set RDB$DESCRIPTION = 'Student''s Identification Number'  where RDB$FIELD_NAME = 'IDNUMBER' and RDB$RELATION_NAME = 'STUDENTS';
UPDATE RDB$RELATION_FIELDS set RDB$DESCRIPTION = 'Student''s Firstname'  where RDB$FIELD_NAME = 'FIRSTNAME' and RDB$RELATION_NAME = 'STUDENTS';
UPDATE RDB$RELATION_FIELDS set RDB$DESCRIPTION = 'Student''s Middlename'  where RDB$FIELD_NAME = 'MIDDLENAME' and RDB$RELATION_NAME = 'STUDENTS';
UPDATE RDB$RELATION_FIELDS set RDB$DESCRIPTION = 'Student''s Lastname'  where RDB$FIELD_NAME = 'LASTNAME' and RDB$RELATION_NAME = 'STUDENTS';
CREATE DESCENDING INDEX IDX_STUDENTS1 ON STUDENTS (MSUSIDNUM);
CREATE DESCENDING INDEX IDX_STUDENTS2 ON STUDENTS (YEARADMITTED);
CREATE INDEX IDX_STUDENTS3 ON STUDENTS (LASTNAME);
CREATE INDEX IDX_STUDENTS4 ON STUDENTS (FIRSTNAME);
CREATE INDEX IDX_STUDENTS5 ON STUDENTS (TERMADMITTED);
CREATE INDEX IDX_STUDENTS6 ON STUDENTS (COURSE);
